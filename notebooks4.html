import pandas as pd
import psycopg2
import numpy as np

iris_df=pd.read_csv("https://raw.githubusercontent.com/HamedAyani114/dataset/main/csv/Iris.csv")

sl = iris_df["SepalLengthCm"]
sw = iris_df["SepalWidthCm"]
pl = iris_df["PetalLengthCm"]
pwcs = iris_df["PetalWidthCm"]
spec = iris_df["Species"]
